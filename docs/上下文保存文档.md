```python
def _create_progressbar(self, comp, layout):
    """创建进度条组件"""
    progressbar = QProgressBar()
    progressbar.setValue(comp.value)
    progressbar.setTextVisible(comp.show_text)
    
    # 如果启用假进度动画
    if comp.auto_progress:
        QTimer.singleShot(500, 
            lambda: self._start_fake_progress(progressbar, comp.duration, comp.signals, comp.id))
    
    return progressbar

def _start_fake_progress(self, progressbar, duration, signals, comp_id):
    """启动假进度动画"""
    update_interval = 50  # 毫秒
    total_steps = (duration * 1000) // update_interval
    step_value = 100 / total_steps
    
    current_step = [0]
    
    def update_progress():
        current_step[0] += 1
        new_value = int(current_step[0] * step_value)
        
        if new_value >= 100:
            progressbar.setValue(100)
            timer.stop()
            
            # 触发信号跳转
            for signal in signals:
                if signal.get('signal_type') == 'progress_complete':
                    target_window_id = signal.get('target_window_id')
                    if target_window_id:
                        QTimer.singleShot(200, 
                            lambda twid=target_window_id: self._open_window(twid))
        else:
            progressbar.setValue(new_value)
    
    timer = QTimer()
    timer.timeout.connect(update_progress)
    timer.start(update_interval)
```
