**修改2：添加假进度动画实现（第332-393行）**
```python
def _start_fake_progress(self, progressbar, duration, signals, comp_id):
    """启动假进度动画
    
    Args:
        progressbar: 进度条控件
        duration: 持续时间（秒）
        signals: 信号列表
        comp_id: 组件ID
    """
    # 计算更新间隔（每秒更新20次，即50毫秒）
    update_interval = 50  # 毫秒
    total_steps = (duration * 1000) // update_interval  # 总步数
    step_value = 100 / total_steps  # 每步增加的值
    
    current_step = [0]  # 使用列表保存当前步数（闭包中可修改）
    
    def update_progress():
        current_step[0] += 1
        new_value = int(current_step[0] * step_value)
        
        if new_value >= 100:
            progressbar.setValue(100)
            timer.stop()
            
            # 【触发信号】进度条完成，检查是否有信号需要触发
            for signal in signals:
                if signal.get('signal_type') == 'progress_complete':
                    target_window_id = signal.get('target_window_id', '')
                    if target_window_id:
                        # 延迟200毫秒后跳转，让用户看到100%
                        QTimer.singleShot(200, 
                            lambda twid=target_window_id: self._open_window(twid))
        else:
            progressbar.setValue(new_value)
    
    timer = QTimer()
    timer.timeout.connect(update_progress)
    timer.start(update_interval)
```
